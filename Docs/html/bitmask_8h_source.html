77 /* Returns nonzero if the bit at (x,y) is set. Coordinates start at
78  (0,0) */
79 static INLINE int bitmask_getbit(const bitmask_t *m, int x, int y)
80 {
81  return (m->bits[x/BITMASK_W_LEN*m->h + y] & BITMASK_N(x & BITMASK_W_MASK)) != 0;
82 }
83 
84 /* Sets the bit at (x,y) */
85 static INLINE void bitmask_setbit(bitmask_t *m, int x, int y)
86 {
87  m->bits[x/BITMASK_W_LEN*m->h + y] |= BITMASK_N(x & BITMASK_W_MASK);
88 }

90 /* Clears the bit at (x,y) */
91 static INLINE void bitmask_clearbit(bitmask_t *m, int x, int y)
92 {
93  m->bits[x/BITMASK_W_LEN*m->h + y] &= ~BITMASK_N(x & BITMASK_W_MASK);
