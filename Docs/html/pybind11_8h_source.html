420  static PyObject *dispatcher(PyObject *self, PyObject *args_in, PyObject *kwargs_in) {
421  using namespace detail;
422 
423  /* Iterator over the list of potentially admissible overloads */
424  const function_record *overloads = (function_record *) PyCapsule_GetPointer(self, nullptr),
425  *it = overloads;
426 
427  /* Need to know how many arguments + keyword arguments there are to pick the right overload */
428  const size_t n_args_in = (size_t) PyTuple_GET_SIZE(args_in);
429 
430  handle parent = n_args_in > 0 ? PyTuple_GET_ITEM(args_in, 0) : nullptr,
431  result = PYBIND11_TRY_NEXT_OVERLOAD;
432 
433  auto self_value_and_holder = value_and_holder();
434  if (overloads->is_constructor) {
435  const auto tinfo = get_type_info((PyTypeObject *) overloads->scope.ptr());
436  const auto pi = reinterpret_cast<instance *>(parent.ptr());
437  self_value_and_holder = pi->get_value_and_holder(tinfo, false);
438 
439  if (!self_value_and_holder.type || !self_value_and_holder.inst) {
440  PyErr_SetString(PyExc_TypeError, "__init__(self, ...) called with invalid `self` argument");
441  return nullptr;
442  }
443 
444  // If this value is already registered it must mean __init__ is invoked multiple times;
445  // we really can't support that in C++, so just ignore the second __init__.
446  if (self_value_and_holder.instance_registered())
447  return none().release().ptr();
448  }
449 
450  try {
451  // We do this in two passes: in the first pass, we load arguments with `convert=false`;
452  // in the second, we allow conversion (except for arguments with an explicit
453  // py::arg().noconvert()). This lets us prefer calls without conversion, with
454  // conversion as a fallback.
455  std::vector<function_call> second_pass;
456 
457  // However, if there are no overloads, we can just skip the no-convert pass entirely
458  const bool overloaded = it != nullptr && it->next != nullptr;
459 
460  for (; it != nullptr; it = it->next) {
461 
462  /* For each overload:
463  1. Copy all positional arguments we were given, also checking to make sure that
464  named positional arguments weren't *also* specified via kwarg.
465  2. If we weren't given enough, try to make up the omitted ones by checking
466  whether they were provided by a kwarg matching the `py::arg("name")` name. If
467  so, use it (and remove it from kwargs; if not, see if the function binding
468  provided a default that we can use.
469  3. Ensure that either all keyword arguments were "consumed", or that the function
470  takes a kwargs argument to accept unconsumed kwargs.
471  4. Any positional arguments still left get put into a tuple (for args), and any
472  leftover kwargs get put into a dict.
473  5. Pack everything into a vector; if we have py::args or py::kwargs, they are an
474  extra tuple or dict at the end of the positional arguments.
475  6. Call the function call dispatcher (function_record::impl)
476 
477  If one of these fail, move on to the next overload and keep trying until we get a
478  result other than PYBIND11_TRY_NEXT_OVERLOAD.
479  */
480 
481  const function_record &func = *it;
482  size_t pos_args = func.nargs; // Number of positional arguments that we need
483  if (func.has_args) --pos_args; // (but don't count py::args
484  if (func.has_kwargs) --pos_args; // or py::kwargs)
485 
486  if (!func.has_args && n_args_in > pos_args)
487  continue; // Too many arguments for this overload
488 
489  if (n_args_in < pos_args && func.args.size() < pos_args)
490  continue; // Not enough arguments given, and not enough defaults to fill in the blanks
491 
492  function_call call(func, parent);
493 
494  size_t args_to_copy = std::min(pos_args, n_args_in);
495  size_t args_copied = 0;
496 
497  // 0. Inject new-style `self` argument
498  if (func.is_new_style_constructor) {
499  // The `value` may have been preallocated by an old-style `__init__`
500  // if it was a preceding candidate for overload resolution.
501  if (self_value_and_holder)
502  self_value_and_holder.type->dealloc(self_value_and_holder);
503 
504  call.init_self = PyTuple_GET_ITEM(args_in, 0);
505  call.args.push_back(reinterpret_cast<PyObject *>(&self_value_and_holder));
506  call.args_convert.push_back(false);
507  ++args_copied;
508  }
509 
510  // 1. Copy any position arguments given.
511  bool bad_arg = false;
512  for (; args_copied < args_to_copy; ++args_copied) {
513  const argument_record *arg_rec = args_copied < func.args.size() ? &func.args[args_copied] : nullptr;
514  if (kwargs_in && arg_rec && arg_rec->name && PyDict_GetItemString(kwargs_in, arg_rec->name)) {
515  bad_arg = true;
516  break;
517  }
518 
519  handle arg(PyTuple_GET_ITEM(args_in, args_copied));
520  if (arg_rec && !arg_rec->none && arg.is_none()) {
521  bad_arg = true;
522  break;
523  }
524  call.args.push_back(arg);
525  call.args_convert.push_back(arg_rec ? arg_rec->convert : true);
526  }
527  if (bad_arg)
528  continue; // Maybe it was meant for another overload (issue #688)
529 
530  // We'll need to copy this if we steal some kwargs for defaults
531  dict kwargs = reinterpret_borrow<dict>(kwargs_in);
532 
533  // 2. Check kwargs and, failing that, defaults that may help complete the list
534  if (args_copied < pos_args) {
535  bool copied_kwargs = false;
536 
537  for (; args_copied < pos_args; ++args_copied) {
538  const auto &arg = func.args[args_copied];
539 
540  handle value;
541  if (kwargs_in && arg.name)
542  value = PyDict_GetItemString(kwargs.ptr(), arg.name);
543 
544  if (value) {
545  // Consume a kwargs value
546  if (!copied_kwargs) {
547  kwargs = reinterpret_steal<dict>(PyDict_Copy(kwargs.ptr()));
548  copied_kwargs = true;
549  }
550  PyDict_DelItemString(kwargs.ptr(), arg.name);
551  } else if (arg.value) {
552  value = arg.value;
553  }
554 
555  if (value) {
556  call.args.push_back(value);
557  call.args_convert.push_back(arg.convert);
558  }
559  else
560  break;
561  }
562 
563  if (args_copied < pos_args)
564  continue; // Not enough arguments, defaults, or kwargs to fill the positional arguments
565  }
566 
567  // 3. Check everything was consumed (unless we have a kwargs arg)
568  if (kwargs && kwargs.size() > 0 && !func.has_kwargs)
569  continue; // Unconsumed kwargs, but no py::kwargs argument to accept them
570 
571  // 4a. If we have a py::args argument, create a new tuple with leftovers
572  if (func.has_args) {
573  tuple extra_args;
574  if (args_to_copy == 0) {
575  // We didn't copy out any position arguments from the args_in tuple, so we
576  // can reuse it directly without copying:
577  extra_args = reinterpret_borrow<tuple>(args_in);
578  } else if (args_copied >= n_args_in) {
579  extra_args = tuple(0);
580  } else {
581  size_t args_size = n_args_in - args_copied;
582  extra_args = tuple(args_size);
583  for (size_t i = 0; i < args_size; ++i) {
584  extra_args[i] = PyTuple_GET_ITEM(args_in, args_copied + i);
585  }
586  }
587  call.args.push_back(extra_args);
588  call.args_convert.push_back(false);
589  call.args_ref = std::move(extra_args);
590  }
591 
592  // 4b. If we have a py::kwargs, pass on any remaining kwargs
593  if (func.has_kwargs) {
594  if (!kwargs.ptr())
595  kwargs = dict(); // If we didn't get one, send an empty one
596  call.args.push_back(kwargs);
597  call.args_convert.push_back(false);
598  call.kwargs_ref = std::move(kwargs);
599  }
600 
601  // 5. Put everything in a vector. Not technically step 5, we've been building it
602  // in `call.args` all along.
603  #if !defined(NDEBUG)
604  if (call.args.size() != func.nargs || call.args_convert.size() != func.nargs)
605  pybind11_fail("Internal error: function call dispatcher inserted wrong number of arguments!");
606  #endif
607 
608  std::vector<bool> second_pass_convert;
609  if (overloaded) {
610  // We're in the first no-convert pass, so swap out the conversion flags for a
611  // set of all-false flags. If the call fails, we'll swap the flags back in for
612  // the conversion-allowed call below.
613  second_pass_convert.resize(func.nargs, false);
614  call.args_convert.swap(second_pass_convert);
615  }
616 
617  // 6. Call the function.
618  try {
619  loader_life_support guard{};
620  result = func.impl(call);
621  } catch (reference_cast_error &) {
622  result = PYBIND11_TRY_NEXT_OVERLOAD;
623  }
624 
625  if (result.ptr() != PYBIND11_TRY_NEXT_OVERLOAD)
626  break;
627 
628  if (overloaded) {
629  // The (overloaded) call failed; if the call has at least one argument that
630  // permits conversion (i.e. it hasn't been explicitly specified `.noconvert()`)
631  // then add this call to the list of second pass overloads to try.
632  for (size_t i = func.is_method ? 1 : 0; i < pos_args; i++) {
633  if (second_pass_convert[i]) {
634  // Found one: swap the converting flags back in and store the call for
635  // the second pass.
636  call.args_convert.swap(second_pass_convert);
637  second_pass.push_back(std::move(call));
638  break;
639  }
640  }
641  }
642  }
643 
644  if (overloaded && !second_pass.empty() && result.ptr() == PYBIND11_TRY_NEXT_OVERLOAD) {
645  // The no-conversion pass finished without success, try again with conversion allowed
646  for (auto &call : second_pass) {
647  try {
648  loader_life_support guard{};
649  result = call.func.impl(call);
650  } catch (reference_cast_error &) {
651  result = PYBIND11_TRY_NEXT_OVERLOAD;
652  }
653 
654  if (result.ptr() != PYBIND11_TRY_NEXT_OVERLOAD) {
655  // The error reporting logic below expects 'it' to be valid, as it would be
656  // if we'd encountered this failure in the first-pass loop.
657  if (!result)
658  it = &call.func;
659  break;
660  }
661  }
662  }
663  } catch (error_already_set &e) {
664  e.restore();
665  return nullptr;
666  } catch (...) {
667  /* When an exception is caught, give each registered exception
668  translator a chance to translate it to a Python exception
669  in reverse order of registration.
670 
671  A translator may choose to do one of the following:
672 
673  - catch the exception and call PyErr_SetString or PyErr_SetObject
674  to set a standard (or custom) Python exception, or
675  - do nothing and let the exception fall through to the next translator, or
676  - delegate translation to the next translator by throwing a new type of exception. */
677 
678  auto last_exception = std::current_exception();
679  auto &registered_exception_translators = get_internals().registered_exception_translators;
680  for (auto& translator : registered_exception_translators) {
681  try {
682  translator(last_exception);
683  } catch (...) {
684  last_exception = std::current_exception();
685  continue;
686  }
687  return nullptr;
688  }
689  PyErr_SetString(PyExc_SystemError, "Exception escaped from default exception translator!");
690  return nullptr;
691  }
692 
693  auto append_note_if_missing_header_is_suspected = [](std::string &msg) {
694  if (msg.find("std::") != std::string::npos) {
695  msg += "\n\n"
696  "Did you forget to `#include <pybind11/stl.h>`? Or <pybind11/complex.h>,\n"
697  "<pybind11/functional.h>, <pybind11/chrono.h>, etc. Some automatic\n"
698  "conversions are optional and require extra headers to be included\n"
699  "when compiling your pybind11 module.";
700  }
701  };
702 
703  if (result.ptr() == PYBIND11_TRY_NEXT_OVERLOAD) {
704  if (overloads->is_operator)
705  return handle(Py_NotImplemented).inc_ref().ptr();
706 
707  std::string msg = std::string(overloads->name) + "(): incompatible " +
708  std::string(overloads->is_constructor ? "constructor" : "function") +

710 
711  int ctr = 0;

713  msg += " "+ std::to_string(++ctr) + ". ";
714 
715  bool wrote_sig = false;
716  if (overloads->is_constructor) {
717  // For a constructor, rewrite `(self: Object, arg0, ...) -> NoneType` as `Object(arg0, ...)`
718  std::string sig = it2->signature;
719  size_t start = sig.find('(') + 7; // skip "(self: "
720  if (start < sig.size()) {
721  // End at the , for the next argument
722  size_t end = sig.find(", "), next = end + 2;
723  size_t ret = sig.rfind(" -> ");
724  // Or the ), if there is no comma:
725  if (end >= sig.size()) next = end = sig.find(')');
726  if (start < end && next < sig.size()) {
727  msg.append(sig, start, end - start);
728  msg += '(';
729  msg.append(sig, next, ret - next);
730  wrote_sig = true;
731  }
732  }
733  }
734  if (!wrote_sig) msg += it2->signature;
735 
736  msg += "\n";
737  }
738  msg += "\nInvoked with: ";
739  auto args_ = reinterpret_borrow<tuple>(args_in);
740  bool some_args = false;
741  for (size_t ti = overloads->is_constructor ? 1 : 0; ti < args_.size(); ++ti) {
742  if (!some_args) some_args = true;
743  else msg += ", ";
744  msg += pybind11::repr(args_[ti]);
745  }
746  if (kwargs_in) {
747  auto kwargs = reinterpret_borrow<dict>(kwargs_in);
748  if (kwargs.size() > 0) {
749  if (some_args) msg += "; ";
750  msg += "kwargs: ";
751  bool first = true;
752  for (auto kwarg : kwargs) {
753  if (first) first = false;
754  else msg += ", ";
755  msg += pybind11::str("{}={!r}").format(kwarg.first, kwarg.second);
756  }
757  }
758  }
759 
760  append_note_if_missing_header_is_suspected(msg);
761  PyErr_SetString(PyExc_TypeError, msg.c_str());
762  return nullptr;
763  } else if (!result) {
764  std::string msg = "Unable to convert function return value to a "
765  "Python type! The signature was\n\t";
766  msg += it->signature;
767  append_note_if_missing_header_is_suspected(msg);
768  PyErr_SetString(PyExc_TypeError, msg.c_str());
769  return nullptr;
770  } else {
771  if (overloads->is_constructor && !self_value_and_holder.holder_constructed()) {
772  auto *pi = reinterpret_cast<instance *>(parent.ptr());
773  self_value_and_holder.type->init_instance(pi, nullptr);
774  }
775  return result.ptr();
776  }
