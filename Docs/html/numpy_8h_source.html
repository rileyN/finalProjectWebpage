310 protected:
311  static constexpr bool Dynamic = Dims < 0;
312  const unsigned char *data_;
313  // Storing the shape & strides in local variables (i.e. these arrays) allows the compiler to
314  // make large performance gains on big, nested loops, but requires compile-time dimensions
315  conditional_t<Dynamic, const ssize_t *, std::array<ssize_t, (size_t) Dims>>
316  shape_, strides_;
317  const ssize_t dims_;
318 
319  friend class pybind11::array;
320  // Constructor for compile-time dimensions:
321  template <bool Dyn = Dynamic>
322  unchecked_reference(const void *data, const ssize_t *shape, const ssize_t *strides, enable_if_t<!Dyn, ssize_t>)
323  : data_{reinterpret_cast<const unsigned char *>(data)}, dims_{Dims} {
324  for (size_t i = 0; i < (size_t) dims_; i++) {
325  shape_[i] = shape[i];
326  strides_[i] = strides[i];
327  }
328  }
329  // Constructor for runtime dimensions:
330  template <bool Dyn = Dynamic>
331  unchecked_reference(const void *data, const ssize_t *shape, const ssize_t *strides, enable_if_t<Dyn, ssize_t> dims)
332  : data_{reinterpret_cast<const unsigned char *>(data)}, shape_{shape}, strides_{strides}, dims_{dims} {}

334 public:
340  template <typename... Ix> const T &operator()(Ix... index) const {
341  static_assert(ssize_t{sizeof...(Ix)} == Dims || Dynamic,
342  "Invalid number of indices for unchecked array reference");
343  return *reinterpret_cast<const T *>(data_ + byte_offset_unsafe(strides_, ssize_t(index)...));
344  }
349  template <ssize_t D = Dims, typename = enable_if_t<D == 1 || Dynamic>>
350  const T &operator[](ssize_t index) const { return operator()(index); }

475  dtype strip_padding(ssize_t itemsize) {
476  // Recursively strip all void fields with empty names that are generated for
477  // padding fields (as of NumPy v1.11).
478  if (!has_fields())
479  return *this;
480 
481  struct field_descr { PYBIND11_STR_TYPE name; object format; pybind11::int_ offset; };
482  std::vector<field_descr> field_descriptors;
483 
484  for (auto field : attr("fields").attr("items")()) {
485  auto spec = field.cast<tuple>();
486  auto name = spec[0].cast<pybind11::str>();
487  auto format = spec[1].cast<tuple>()[0].cast<dtype>();
488  auto offset = spec[1].cast<tuple>()[1].cast<pybind11::int_>();
489  if (!len(name) && format.kind() == 'V')
490  continue;
491  field_descriptors.push_back({(PYBIND11_STR_TYPE) name, format.strip_padding(format.itemsize()), offset});
492  }
493 
494  std::sort(field_descriptors.begin(), field_descriptors.end(),
495  [](const field_descr& a, const field_descr& b) {
496  return a.offset.cast<int>() < b.offset.cast<int>();
497  });
498 
499  list names, formats, offsets;
500  for (auto& descr : field_descriptors) {
501  names.append(descr.name);
502  formats.append(descr.format);
503  offsets.append(descr.offset);
504  }
505  return dtype(names, formats, offsets, itemsize);

1060 inline PYBIND11_NOINLINE void register_structured_dtype(
1061  any_container<field_descriptor> fields,
1062  const std::type_info& tinfo, ssize_t itemsize,
1063  bool (*direct_converter)(PyObject *, void *&)) {
1064 
1065  auto& numpy_internals = get_numpy_internals();
1066  if (numpy_internals.get_type_info(tinfo, false))
1067  pybind11_fail("NumPy: dtype is already registered");
1068 
1069  list names, formats, offsets;
1070  for (auto field : *fields) {
1071  if (!field.descr)
1072  pybind11_fail(std::string("NumPy: unsupported field dtype: `") +
1073  field.name + "` @ " + tinfo.name());
1074  names.append(PYBIND11_STR_TYPE(field.name));
1075  formats.append(field.descr);
1076  offsets.append(pybind11::int_(field.offset));
1077  }
1078  auto dtype_ptr = pybind11::dtype(names, formats, offsets, itemsize).release().ptr();
1079 
1080  // There is an existing bug in NumPy (as of v1.11): trailing bytes are
1081  // not encoded explicitly into the format string. This will supposedly
1082  // get fixed in v1.12; for further details, see these:
1083  // - https://github.com/numpy/numpy/issues/7797
1084  // - https://github.com/numpy/numpy/pull/7798
1085  // Because of this, we won't use numpy's logic to generate buffer format
1086  // strings and will just do it ourselves.
1087  std::vector<field_descriptor> ordered_fields(std::move(fields));
1088  std::sort(ordered_fields.begin(), ordered_fields.end(),
1089  [](const field_descriptor &a, const field_descriptor &b) { return a.offset < b.offset; });
1090  ssize_t offset = 0;
1091  std::ostringstream oss;
1092  // mark the structure as unaligned with '^', because numpy and C++ don't
1093  // always agree about alignment (particularly for complex), and we're
1094  // explicitly listing all our padding. This depends on none of the fields
1095  // overriding the endianness. Putting the ^ in front of individual fields
1096  // isn't guaranteed to work due to https://github.com/numpy/numpy/issues/9049
1097  oss << "^T{";
1098  for (auto& field : ordered_fields) {
1099  if (field.offset > offset)
1100  oss << (field.offset - offset) << 'x';
1101  oss << field.format << ':' << field.name << ':';
1102  offset = field.offset + field.size;
1103  }
1104  if (itemsize > offset)
1105  oss << (itemsize - offset) << 'x';
1106  oss << '}';
1107  auto format_str = oss.str();
1108 
1109  // Sanity check: verify that NumPy properly parses our buffer format string
1110  auto& api = npy_api::get();
1111  auto arr = array(buffer_info(nullptr, itemsize, format_str, 1));
1112  if (!api.PyArray_EquivTypes_(dtype_ptr, arr.dtype().ptr()))
1113  pybind11_fail("NumPy: invalid buffer descriptor!");
1114 
1115  auto tindex = std::type_index(tinfo);
1116  numpy_internals.registered_dtypes[tindex] = { dtype_ptr, format_str };
1117  get_internals().direct_conversions[tindex].push_back(direct_converter);
1118 }

1355 // Populates the shape and number of dimensions for the set of buffers. Returns a broadcast_trivial
1356 // enum value indicating whether the broadcast is "trivial"--that is, has each buffer being either a
1357 // singleton or a full-size, C-contiguous (`c_trivial`) or Fortran-contiguous (`f_trivial`) storage
1358 // buffer; returns `non_trivial` otherwise.
1359 template <size_t N>
1360 broadcast_trivial broadcast(const std::array<buffer_info, N> &buffers, ssize_t &ndim, std::vector<ssize_t> &shape) {
1361  ndim = std::accumulate(buffers.begin(), buffers.end(), ssize_t(0), [](ssize_t res, const buffer_info &buf) {
1362  return std::max(res, buf.ndim);
1363  });
1364 
1365  shape.clear();
1366  shape.resize((size_t) ndim, 1);
1367 
1368  // Figure out the output size, and make sure all input arrays conform (i.e. are either size 1 or
1369  // the full size).
1370  for (size_t i = 0; i < N; ++i) {
1371  auto res_iter = shape.rbegin();
1372  auto end = buffers[i].shape.rend();
1373  for (auto shape_iter = buffers[i].shape.rbegin(); shape_iter != end; ++shape_iter, ++res_iter) {
1374  const auto &dim_size_in = *shape_iter;
1375  auto &dim_size_out = *res_iter;
1376 
1377  // Each input dimension can either be 1 or `n`, but `n` values must match across buffers
1378  if (dim_size_out == 1)
1379  dim_size_out = dim_size_in;
1380  else if (dim_size_in != 1 && dim_size_in != dim_size_out)
1381  pybind11_fail("pybind11::vectorize: incompatible size/dimension of inputs!");
1382  }
1383  }
1384 
1385  bool trivial_broadcast_c = true;
1386  bool trivial_broadcast_f = true;
1387  for (size_t i = 0; i < N && (trivial_broadcast_c || trivial_broadcast_f); ++i) {
1388  if (buffers[i].size == 1)
1389  continue;
1390 
1391  // Require the same number of dimensions:
1392  if (buffers[i].ndim != ndim)
1393  return broadcast_trivial::non_trivial;
1394 
1395  // Require all dimensions be full-size:
1396  if (!std::equal(buffers[i].shape.cbegin(), buffers[i].shape.cend(), shape.cbegin()))
1397  return broadcast_trivial::non_trivial;
1398 
1399  // Check for C contiguity (but only if previous inputs were also C contiguous)
1400  if (trivial_broadcast_c) {
1401  ssize_t expect_stride = buffers[i].itemsize;
1402  auto end = buffers[i].shape.crend();
1403  for (auto shape_iter = buffers[i].shape.crbegin(), stride_iter = buffers[i].strides.crbegin();
1404  trivial_broadcast_c && shape_iter != end; ++shape_iter, ++stride_iter) {
1405  if (expect_stride == *stride_iter)
1406  expect_stride *= *shape_iter;
1407  else
1408  trivial_broadcast_c = false;
1409  }
1410  }
1411 
1412  // Check for Fortran contiguity (if previous inputs were also F contiguous)
1413  if (trivial_broadcast_f) {
1414  ssize_t expect_stride = buffers[i].itemsize;
1415  auto end = buffers[i].shape.cend();
1416  for (auto shape_iter = buffers[i].shape.cbegin(), stride_iter = buffers[i].strides.cbegin();
1417  trivial_broadcast_f && shape_iter != end; ++shape_iter, ++stride_iter) {
1418  if (expect_stride == *stride_iter)
1419  expect_stride *= *shape_iter;
1420  else
1421  trivial_broadcast_f = false;
1422  }
1423  }
1424  }
1425 
1426  return
1427  trivial_broadcast_c ? broadcast_trivial::c_trivial :
1428  trivial_broadcast_f ? broadcast_trivial::f_trivial :
1429  broadcast_trivial::non_trivial;
